\section{Conclusion}

\begin{frame}{Discussion}
    \begin{itemize}
        \item no significant difference in recall for the methods
        \item[→] hypothesis that HOT MSAX has higher recall than HOT SAX cannot 
            be supported
        \item MSAX has smallest alphabet size and highest dimension reduction
        \item[→] MSAX may be more efficient in achieving same results as S-SAX,
            D-SAX
        \item \citeauthor*{anacleto2020} showed similar D-SAX vs MSAX 
            performance which supports this work's conclusion
    \end{itemize}
\end{frame}

\begin{frame}{Novel Contributions}
    \begin{itemize}
        \item application of MSAX to ECG discord discovery and medical data in
            general
        \item the HOT MSAX algorithm, a modification of HOT SAX that uses MSAX
        \item the expansion of HOT SAX to multivariate time series through HOT
            MSAX
    \end{itemize}
\end{frame}

\begin{frame}{Conclusion}
    \begin{itemize}
        \item best parameters for S-SAX, D-SAX, MSAX were found
        \item could not demonstrate higher recall of HOT MSAX for best 
            parameters
        \item showed viability of a discord classifier to assist with ECG 
            analysis
        \item future research possibilities: use different ECG data, different 
            detection criteria
    \end{itemize}
\end{frame}

\subsection*{Preliminary Results}

\begin{frame}{Preliminary Results}
    \begin{itemize}
        \item compared SAX and MSAX using the top $k=80$ discords (816 sets of
            discords total)
        \item analyzed the relevance of results with recall (sensitivity)
            \mynote{how many relevant items are selected}
            \mynote{recall = true positive / (true positive + false negative)}
            \mynote{this is done because for medical things it is more useful
            to look at a couple too many segments than not enough}
        \item recall for MSAX is higher compared to SAX
        \item if SAX is applied to two leads and the results combined, it
            slightly outperforms MSAX
    \end{itemize}
\end{frame}
