\section[Results]{Preliminary Results}

\subsection*{Implementation}

\begin{frame}{Implementation}
    \begin{itemize}
        \item SAX, MSAX, HOTSAX was implemented in Julia (scientific
            programming language)
            \mynote{fast, type support, great libraries, JIT compilation}
        \item used annotated digital ECGs from the MIT-BIH arrhythmia database
            \mycite{moody1992}
            \mynote{ecgs have all heart beats annotated}
            \mynote{know which are normal, diseases, noise, etc}
            \mynote{48 recordings of 30 minutes}
        \item HOTSAX was performed for different $w$, $B$, subsequence lengths
            \mynote{$w$ - paa segments; $B$ - alphabet size; subsequence
            length for HOTSAX}
        \item results were exported to CSV file and analyzed using the
            R programming language
    \end{itemize}
\end{frame}

\subsection*{Preliminary Results}

\begin{frame}{Preliminary Results}
    \begin{itemize}
        \item compared SAX and MSAX using the top $k=80$ discords (816 sets of
            discords total)
        \item analyzed the relevance of results with recall (sensitivity)
            \mynote{how many relevant items are selected}
            \mynote{recall = true positive / (true positive + false negative)}
            \mynote{this is done because for medical things it is more useful
            to look at a couple too many segments than not enough}
        \item recall for MSAX is higher compared to SAX
        \item if SAX is applied to two leads and the results combined, it
            slightly outperforms MSAX
    \end{itemize}
\end{frame}
