\begin{frame}{Method Background}
    \begin{itemize}
        \item \citeauthor*{lin2003} (\citeyear{lin2003}):\\

            Heuristically Ordered Time series using SAX (HOT SAX): discord
            discovery algorithm using SAX\mycite{keogh2005}
            \mynote{uses sax representation to make the finding of discords
            easier}
            \mynote{can use MSAX just as well}
        \item \citeauthor*{anacleto2020} (\citeyear{anacleto2020}):\\
            Multivariate SAX (MSAX): expands SAX to multivariate time series
            \mycite{anacleto2020}
            \mynote{takes the correlation between ecg leads into account}
            \mynote{cov mat: covariance between each lead and variance on diag}
    \end{itemize}
\end{frame}

\begin{frame}{SAX and MSAX -- Distance Measure}
    \begin{itemize}
        \item needed to compare two SAX/MSAX segments
        \item sum of distances between symbols
        \item symbol distance is based on difference of breakpoints
        \item lower-bounds Euclidean Distance, i.e. corresponds to ``real" 
            distance
    \end{itemize}
\end{frame}

\subsection*{HOT SAX and HOT MSAX}

% TODO: add some points from next slide
\begin{frame}{HOT SAX and HOT MSAX -- Overview}
    \begin{itemize}
        \item HOT SAX: find discords in SAX-represented time series
        \item classifies time series segments into ``discord" and
            ``non-discord"
        \item HOT MSAX: uses MSAX instead of SAX
        \item HOT MSAX can work with multivariate time series
    \end{itemize}
\end{frame}

% TODO: move to appendix
\begin{frame}{HOT SAX and HOT MSAX -- Heuristic}
    \begin{itemize}
        \item two parameters: $m$ and $k$
        \item two assumptions:
            \begin{itemize}[label=--]
                \item time series discords are rare
                \item segments similar to discords may also be discords
            \end{itemize}
        \item speed up discord discovery:
            \begin{itemize}[label=--]
                \item consider rarest segments first
                \item consider similar segments together
            \end{itemize}
    \end{itemize}
\end{frame}

\subsection*{Research Process}

\begin{frame}{Analysis Process}
    \begin{enumerate}[label=(\arabic*)]
        \item perform HOT SAX and HOT MSAX for many parameter combinations for
            all ECGs in the MIT-BIH database
        \item find recall, precision for each combination
        \item set recall threshold of 95\%, then sort by precision
        \item choose top 10 of those parameters for each method
        \item choose best parameters for each method using box plot, 
            interquartile range, outliers
    \end{enumerate}
\end{frame}

\begin{frame}{Three Datasets}
    \begin{enumerate}[label=(\arabic*),itemsep=1.5em]
        \item S-SAX: data for HOT SAX algorithm, considering each lead
            separately
        \item D-SAX: data for HOT SAX, considering both leads combined
        \item MSAX: data for the HOT MSAX algorithm
    \end{enumerate}
\end{frame}
