\subsection*{HOT SAX and HOT MSAX}

\begin{frame}{HOT SAX and HOT MSAX -- Overview}
    \begin{itemize}
        \item \hyperlink{s-hs}{HOT SAX: find discords in SAX-represented time 
            series}
        \item classifies time series segments into ``discord" and
            ``non-discord"
        \item HOT MSAX: uses MSAX instead of SAX
        \item HOT MSAX can work with multivariate time series
        \item \hyperlink{s-hs}{simple heuristic}
        \item only two parameters
    \end{itemize}
\end{frame}

\subsection*{Research Process}

\begin{frame}{Analysis Process}
    \begin{enumerate}[label=(\arabic*)]
        \item perform HOT SAX and HOT MSAX for many parameter combinations for
            all ECGs in the MIT-BIH database
        \item \hyperlink{s-stats}{find recall, precision for each combination}
        \item set recall threshold of 95\%, then sort by precision
        \item choose top 10 of those parameters for each method
        \item choose best parameters for each method using box plot, 
            interquartile range, outliers
    \end{enumerate}
\end{frame}

\begin{frame}{Three Datasets}
    \begin{enumerate}[label=(\arabic*),itemsep=1.5em]
        \item S-SAX: data for HOT SAX algorithm, considering each lead
            separately
        \item D-SAX: data for HOT SAX, considering both leads combined
        \item MSAX: data for the HOT MSAX algorithm
    \end{enumerate}
\end{frame}
