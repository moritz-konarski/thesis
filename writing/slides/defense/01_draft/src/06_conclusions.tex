\section{Conclusions}

\begin{frame}{Discussion}
    \begin{itemize}
        \item no statistically significant difference in recall for the methods
        \item[$\rightarrow$] hypothesis cannot be supported
        \item MSAX has both the smallest alphabet size and highest dimension
            reduction
        \item[$\rightarrow$] this points to MSAX being more efficient in
            achieving the same results 
        \item \textcite{anacleto2020} showed similar performance for ECG
            classification, supports this work's conclusion
    \end{itemize}
\end{frame}

\begin{frame}{Conclusion}
    \begin{itemize}
        \item could not demonstrate superiority of HOT MSAX for best parameters
        \item contributed the HOT MSAX method to literature
        \item HOT MSAX was found to be more efficient than HOT SAX
        \item showed viability of a discord classifier for ECG analysis
    \end{itemize}
\end{frame}

\subsection*{Preliminary Results}

\begin{frame}{Preliminary Results}
    \begin{itemize}
        \item compared SAX and MSAX using the top $k=80$ discords (816 sets of
            discords total)
        \item analyzed the relevance of results with recall (sensitivity)
            \mynote{how many relevant items are selected}
            \mynote{recall = true positive / (true positive + false negative)}
            \mynote{this is done because for medical things it is more useful
            to look at a couple too many segments than not enough}
        \item recall for MSAX is higher compared to SAX
        \item if SAX is applied to two leads and the results combined, it
            slightly outperforms MSAX
    \end{itemize}
\end{frame}
